class Solution {
    public int[] solution(int n) {
        int count = 0;
        int j = 0;
        
        for(int i = 1; i <= n; i++){
            count += n % i == 0 ? 1 : 0;
        }

        int[] answer = new int[count];

        for(int i = 1; i <= n; i++){
            if(n % i == 0){
                answer[j] = i;
                j++;
            }
        }
        return answer;
    }
}
