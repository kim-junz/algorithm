class Solution {
    public int[] solution(int[] arr) {
      int[] answer = {};

        int min = arr[0];
        int k = 0;

        if(arr.length == 1){
            answer = new int[] {-1};

        }else if(arr.length > 1) {
            answer = new int[arr.length - 1];

            for (int i = 0; i < arr.length; i++) {
                if (arr[i] < min) {
                    min = arr[i];
                }
            }
            for (int j = 0; j < arr.length; j++) {
                if (arr[j] == min){
                    continue;
                }
                answer[k++] = arr[j];
            }
        }
        return answer;
    }
}
